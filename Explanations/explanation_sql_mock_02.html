question 08
CREATE TABLE T1
(
C1 INT,
C2 INT
)
GO
ALTER TABLE T1
ADD CHECK (C1 < 100)
--This will work as Table has no rows
GO
INSERT INTO T1 VALUES (1, 50), (2, 60)
GO
SELECT * FROM T1
GO
--Now table has two rows
GO
ALTER TABLE T1
ADD CHECK (C2 > 100)
GO
--This will fail as C2 has colum values 50 in Row 1 and 60
--in Row 2 which are not valid if we set C2 > 100
--
ALTER TABLE T1
WITH NOCHECK
ADD CHECK (C2 > 100)
--This will work, we are telling here "Do not check the data which are in
--the table. But if you add new row after applying the check contraint,
--You must give value for C2 more than 100
GO
INSERT INTO T1 VALUES(3, 110) --ok
INSERT INTO T1 VALUES(3, 70) --fails
GO
SELECT * FROM T1
GO

question 10
CREATE TABLE PK1
(
pkid INT PRIMARY KEY,
pname VARCHAR(30) NOT NULL
)
GO
/*
* SYNTAX
* REFRENCES table-name (colimn-name) ON UPDATE SET DEFAULT/NULL/NO ACTION/CASCADE ON DELETE SET DEFAULT/NULL/NO ACTION/CASCADE
* DEFAULT NO ACTION -- means reject the action and raise error
* If you do not add no rule, just REFRENCES table-name (colimn-name) default rule (NO ACTION) is assumed
* */

CREATE TABLE FK1
(
fkid INT PRIMARY KEY,
points INT NOT NULL,
pkid INT NOT NULL REFERENCES PK1(pkid) ON DELETE NO ACTION
)
GO
INSERT INTO PK1 VALUES (1, 'A'), (2, 'B')
GO
INSERT INTO FK1 VALUES (1, 20, 1) --Ok
INSERT INTO FK1 VALUES (1, 20, 10) --Invalid FK, error will be raised
GO
CREATE TABLE PK2
(
